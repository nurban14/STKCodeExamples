%% Extracting Specific Grouped Time Independent Data
% IAgStkObjectRoot root: STK Object Model root
% IAgFacility facility: Facility object

% Change DateFormat dimension to epoch seconds to make the data easier to handle in MATLAB
root.UnitPreferences.Item('DateFormat').SetCurrentUnit('EpSec');

%Retrieve only Normal x, Normal y, and Normal z datasets from Data providers
elems = {'Normal x';'Normal y'; 'Normal z'};
facDP = facility.DataProviders.Item('Terrain Info').Group.Item('Topocentric').ExecElements(elems);
normX = cell2mat(facDP.DataSets.GetDataSetByName('Normal x').GetValues);
normY = cell2mat(facDP.DataSets.GetDataSetByName('Normal y').GetValues);
normZ = cell2mat(facDP.DataSets.GetDataSetByName('Normal z').GetValues);

%Alternatively...
facDP = facility.DataProviders.GetDataPrvFixedFromPath('Terrain Info/Topocentric').ExecElements(elems);
normX = cell2mat(facDP.DataSets.GetDataSetByName('Normal x').GetValues);
normY = cell2mat(facDP.DataSets.GetDataSetByName('Normal y').GetValues);
normZ = cell2mat(facDP.DataSets.GetDataSetByName('Normal z').GetValues);
