## Extracting All Ungrouped Time Dependent Data separated by Sections
# IAgStkObjectRoot root: STK Object Model root
# IAgScenario scenario: STK Current Scenario
# IAgSatellite satellite: Satellite object

# Change DateFormat dimension to epoch seconds to make the data easier to handle 
root.UnitPreferences.Item('DateFormat').SetCurrentUnit('EpSec')

root.UnitPreferences.Item('Power').SetCurrentUnit('W')
solarPanelData = satellite.DataProviders.Item("Solar Panel Power").Exec(scenario.StartTime,scenario.StopTime, 86400)

time = solarPanelData.Intervals.Item(0).DataSets.GetDataSetByName('Time').GetValues()
power = solarPanelData.Intervals.Item(0).DataSets.GetDataSetByName('Power').GetValues()
solarIntensity = solarPanelData.Intervals.Item(0).DataSets.GetDataSetByName('Solar Intensity').GetValues() 
for i in range(1,solarPanelData.Intervals.Count):
    time = time + solarPanelData.Intervals.Item(i).DataSets.GetDataSetByName('Time').GetValues()
    power = power + solarPanelData.Intervals.Item(i).DataSets.GetDataSetByName('Power').GetValues()
    solarIntensity = solarIntensity + solarPanelData.Intervals.Item(i).DataSets.GetDataSetByName('Solar Intensity').GetValues()
